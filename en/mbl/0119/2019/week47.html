import numpy.linalg as lin

def evec_from_eval(A, i, j):
    lam, v = lin.eig(A)
    n = A.shape[0]
    s = list(range(n))
    s.remove(j)
    o = A[s,:]
    lam2, _ = lin.eig(o[:,s])
    left = np.prod([lam[i] - lam[k] for k in range(n) if k!=i])
    right = np.prod([lam[i] - lam2[k] for k in range(n-1)])
    newval = right/left
    return newval
